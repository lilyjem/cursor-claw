```ts
// src/core/rateLimit/TokenBucket.ts
export interface TokenBucketOptions {
  capacity: number;       // 桶最大 token 数（突发上限）
  refillPerSec: number;   // 每秒回血速率
  now?: () => number;     // 测试可注入 fake clock；默认 Date.now
}

/**
 * 纯算法 token-bucket：可突发到 capacity，稳态按 refillPerSec 速率消耗。
 * 状态只有 (tokens, lastRefillMs)，无 timer / 副作用。
 */
export class TokenBucket {
  private readonly capacity: number;
  private readonly refillPerSec: number;
  private readonly now: () => number;
  private tokens: number;
  private lastRefillMs: number;

  constructor(opts: TokenBucketOptions) {
    this.capacity = opts.capacity;
    this.refillPerSec = opts.refillPerSec;
    this.now = opts.now ?? Date.now;
    this.tokens = opts.capacity;
    this.lastRefillMs = this.now();
  }

  // 同步当前应有的 token 数（基于流逝时间补血）
  private refill(): void {
    const t = this.now();
    const dt = (t - this.lastRefillMs) / 1000;
    if (dt <= 0) return;
    this.tokens = Math.min(this.capacity, this.tokens + dt * this.refillPerSec);
    this.lastRefillMs = t;
  }

  // 试图取 n 个 token；够则扣减返回 true，否则保持原状返回 false
  take(n = 1): boolean {
    this.refill();
    if (this.tokens >= n) {
      this.tokens -= n;
      return true;
    }
    return false;
  }

  // 当前可用 token 数（已 refill）；不消耗
  inspect(): number {
    this.refill();
    return this.tokens;
  }

  // 还需要等多少 ms 才能再 take 1 个 token；满桶时返回 0
  timeUntilNext(): number {
    this.refill();
    if (this.tokens >= 1) return 0;
    const need = 1 - this.tokens;
    return Math.ceil((need / this.refillPerSec) * 1000);
  }
}
```

```ts
// src/core/rateLimit/RateLimiter.ts
import { TokenBucket } from "./TokenBucket.js";

export interface BucketSpec {
  capacity: number;
  refillPerSec: number;
}

export interface RateLimiterOptions {
  // 每种 key 一组 bucket 参数；未声明的 key check() 直接 ALLOW
  buckets: Record<string, BucketSpec>;
  // 内部 (userId, key) → bucket 的最大数；超过就 LRU evict
  maxBuckets?: number;
  now?: () => number;
}

export interface CheckResult {
  allowed: boolean;
  retryAfterMs: number; // ALLOW 时为 0
}

/**
 * 多 bucket 限速容器。
 * - 按 (userId, key) 索引一组独立 TokenBucket
 * - 未在 options.buckets 中声明的 key 视为"无限制"，直接 ALLOW
 * - 内置简易 LRU：lastUsedMs 升序 evict，避免 idle bucket 累积
 */
export class RateLimiter {
  private readonly specs: Record<string, BucketSpec>;
  private readonly maxBuckets: number;
  private readonly now: () => number;
  // 用 Map 保留插入顺序；命中后 delete + set 更新到末尾以模拟 LRU
  private readonly store = new Map<string, TokenBucket>();
  private readonly lastUsed = new Map<string, number>();

  constructor(opts: RateLimiterOptions) {
    this.specs = opts.buckets;
    this.maxBuckets = opts.maxBuckets ?? 1024;
    this.now = opts.now ?? Date.now;
  }

  check(userId: number, key: string): CheckResult {
    const spec = this.specs[key];
    if (!spec) return { allowed: true, retryAfterMs: 0 };

    const id = `${userId}::${key}`;
    let bucket = this.store.get(id);
    if (!bucket) {
      bucket = new TokenBucket({ ...spec, now: this.now });
      this.store.set(id, bucket);
      this.evictIfFull();
    }
    this.lastUsed.set(id, this.now());

    if (bucket.take(1)) return { allowed: true, retryAfterMs: 0 };
    return { allowed: false, retryAfterMs: bucket.timeUntilNext() };
  }

  // 朴素 LRU：按 lastUsed 升序找出最久未用的 bucket 删除
  private evictIfFull(): void {
    if (this.store.size <= this.maxBuckets) return;
    let oldestId: string | undefined;
    let oldestT = Number.POSITIVE_INFINITY;
    for (const [id, t] of this.lastUsed) {
      if (t < oldestT) {
        oldestT = t;
        oldestId = id;
      }
    }
    if (oldestId) {
      this.store.delete(oldestId);
      this.lastUsed.delete(oldestId);
    }
  }
}
```

```ts
// 把 retryAfterMs 渲染成中文用户可读文本。仅做四舍五入到秒，保留 1 位小数避免奇怪的 0 秒。
export function renderRetryAfter(retryMs: number): string {
  const seconds = Math.max(0.1, retryMs / 1000);
  return `${seconds.toFixed(1)} 秒`;
}

export function rateLimitedMessageText(retryMs: number): string {
  return `请求过于频繁，请 ${renderRetryAfter(retryMs)} 后重试。`;
}

export function rateLimitedAgentCreateText(retryMs: number): string {
  return `短时间内创建 agent 过多，请 ${renderRetryAfter(retryMs)} 后重试。`;
}
```

```ts
import type { IMessenger } from "../../core/messenger/IMessenger.js";
import type { RateLimiter } from "../../core/rateLimit/RateLimiter.js";
import { logger } from "../../logger.js";
import { rateLimitedMessageText } from "../../util/rateLimitMessages.js";

/**
 * 入口限速守卫：返回 true 表示允许继续；false 表示已拦截并通知用户。
 * 抽到独立模块是为了让 integration test 可以脱离 main() 直接测。
 */
export async function rateLimitGuard(opts: {
  limiter: RateLimiter;
  messenger: IMessenger;
  chatId: string;
  userId: number;
  key: "msg";
}): Promise<boolean> {
  const r = opts.limiter.check(opts.userId, opts.key);
  if (r.allowed) return true;
  logger.warn(
    { userId: opts.userId, key: opts.key, retryMs: r.retryAfterMs },
    "rate limited",
  );
  await opts.messenger.sendText(opts.chatId, rateLimitedMessageText(r.retryAfterMs), {
    parseMode: "plain",
  });
  return false;
}
```

```ts
// src/core/reminders/ReminderQuota.ts
import type { Reminder, ReminderStore } from "./ReminderStore.js";
import { ReminderQuotaExceededError } from "./errors.js";

export interface ReminderQuotaOptions {
  maxPerUser: number;
}

/**
 * 包装 ReminderStore.add：在写入前检查同 createdBy 的现存数。
 * 不修改 ReminderStore 的契约，保持持久化层职责单一。
 */
export class ReminderQuota {
  constructor(
    private readonly store: Pick<ReminderStore, "list" | "add">,
    private readonly opts: ReminderQuotaOptions,
  ) {}

  async checkAndAdd(item: Reminder): Promise<void> {
    const used = this.store
      .list()
      .filter((r) => r.createdBy === item.createdBy).length;
    if (used >= this.opts.maxPerUser) {
      throw new ReminderQuotaExceededError(used, this.opts.maxPerUser);
    }
    await this.store.add(item);
  }
}
```
